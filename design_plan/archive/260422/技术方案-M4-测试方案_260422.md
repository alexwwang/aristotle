    """Helper: create a reflection record in aristotle-state.json."""
    from aristotle_mcp.config import resolve_repo_dir
    state_path = resolve_repo_dir().parent / "aristotle-state.json"
    records = [{

        "target_session_id": "ses_test123",

    }]
    state_path.parent.mkdir(parents=True, exist_ok=True)
    state_path.write_text(json.dumps(records), encoding="utf-8")

def _create_draft_file(sequence: int, content: str = "## DRAFT Report\nTest content") -> Path:
    """Helper: create a DRAFT file on disk."""
    from aristotle_mcp.config import resolve_repo_dir
    drafts_dir = resolve_repo_dir().parent / "aristotle-drafts"
    drafts_dir.mkdir(parents=True, exist_ok=True)
    path = drafts_dir / f"rec_{sequence}.md"
    path.write_text(content, encoding="utf-8")
    return path

def test_full_reflect_flow(self):

## 八、实施注意事项

### 8.1 私有函数测试策略

`_next_sequence`、`_ensure_repo_initialized`、`_cleanup_stale_workflows` 等以 `_` 开头的函数：
- **优先方案：** 通过公开接口间接测试（如 `orchestrate_start` 内部调用 `_ensure_repo_initialized`）
- **备选方案：** 在 `server.py` 底部添加测试专用导出：
  ```python
  # Test exports (not MCP tools, but accessible via import)
  __test_exports__ = ["_next_sequence", "_ensure_repo_initialized", "_cleanup_stale_workflows"]
  ```

### 8.2 现有测试更新

| 测试 | 变更 | 原因 |
|------|------|------|
| `test_reflect_command_returns_placeholder` | 更新断言 | reflect 从 placeholder 变为真实 `fire_sub` 实现 |
| `test_unknown_event_type_returns_notify` | 更新断言 | `subagent_done` 不再是 unknown（reflect 流使用） |

### 8.3 时间依赖测试

`_cleanup_stale_workflows` 测试需要模拟时间。策略：
- 直接构造带有旧 `updated_at` 的 workflow 文件，不 mock `datetime`
- 使用 ISO 格式字符串 `datetime.now(timezone.utc) - timedelta(hours=25)` 写入

### 8.4 Fixture 复用

新增 helper 函数（`_start_reflect_workflow`、`_fire_r_done_event`、`_setup_reflection_record` 等）定义在 `test_orchestration.py` 模块级，与现有 `_start_learn_workflow`、`_make_verified_rule` 并列。
